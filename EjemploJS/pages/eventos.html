Ejemplo de eventos

A continuación se muestran ejemplos de uso de eventos en JavaScript, incluyendo click, mouseover e input

Código JavaScript


const btnClick = document.getElementById("botonClick");
const cuadroHover cuadro = document.getElementById("cuadroHover");
const inputTexto = document.getElementById("inputTexto");

document.addEventListener("DOMContentLoaded", () => {
    // Evento click
    if(btnClick){
        btnClick.addEventListener("click", () => {
            console.log("Se hizo click en el botón");
        });
    }

    // Evento mouseover
    if(cuadroHover){
        cuadroHover.addEventListener("mouseover", () => {
            console.log("Mouse sobre el cuadro");
        });
    }

    // Evento input
    if(inputTexto){
        inputTexto.addEventListener("input", () => {
            console.log("Texto ingresado:", inputTexto.value);
        });
    }
});
