Ejemplo de condicionales y bucles

A continuación se muestran ejemplos de uso de if/else y bucles for en JavaScript

Código JavaScript


// Condicional if/else
let edad = 20;
let mensaje;
if (edad >= 18) {
    mensaje = "Es mayor de edad";
} else {
    mensaje = "Es menor de edad";
}

// Bucle for
let numeros = "";
for (let i = 1; i <= 5; i++) {
    numeros += i + " ";
}

// Condicional switch
let dia = 3;
let nombreDia;
switch(dia) {
    case 1:
        nombreDia = "Lunes";
        break;
    case 2:
        nombreDia = "Martes";
        break;
    case 3:
        nombreDia = "Miércoles";
        break;
    case 4:
        nombreDia = "Jueves";
        break;
    case 5:
        nombreDia = "Viernes";
        break;
    default:
        nombreDia = "Fin de semana";
        break;
}

console.log(mensaje);
console.log("Números del 1 al 5:", numeros);
console.log("Día de la semana:", nombreDia);
